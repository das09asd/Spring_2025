Процедура ПроверитьСтроку(Команда)
Строка = Объект.ВведенаяСтрока; //Получение введеной строки	
	
	МаксДлина = 0; //Объявление переменной отвевчающей за максимальную длинну последовательности из цифр, идущих подряд. 
    ТекДлина = 0;  //Объявление переменной отвевчающей за текущую длинну последовательности из цифр, идущих подряд.
    
    Для i = 1 По СтрДлина(Строка) Цикл  //цикл обходит каждый символ с i=1 по i=количиство символов строки
        Символ = Сред(Строка, i, 1); //считываем текущий символ
        
        Если Найти("0123456789", Символ) > 0 Тогда  //Проверка текущего символа на цифру
            ТекДлина = ТекДлина + 1;  // если символ цифра записываем результат
            Если ТекДлина > МаксДлина Тогда    // Проверка текущей длинны, если она больше максимальной, перезаписываем максимальную длинну
                МаксДлина = ТекДлина;
            КонецЕсли;
        Иначе
            ТекДлина = 0;  // Если текущий символ не цифра, обнуляем текущую длинну
        КонецЕсли;
    КонецЦикла;
    
    Сообщить("Ответ"+ " "+МаксДлина);  //Вывод  максимальной длинны последовательности из цифр, идущих подряд
